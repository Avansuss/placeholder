cmake_minimum_required(VERSION 3.8)
project(fleet_manager)

# Compiler warnings
if(CMAKE_COMPILER_IS_GNUCXX OR CMAKE_CXX_COMPILER_ID MATCHES "Clang")
  add_compile_options(-Wall -Wextra -Wpedantic)
endif()

# Find ROS 2 packages
find_package(ament_cmake REQUIRED)
find_package(rclcpp REQUIRED)
find_package(std_msgs REQUIRED)

# --- Build executable ---
# executable name = fleet_manager_node
# source file = src/fleet_manager_node.cpp
add_executable(fleet_manager_node src/fleet_manager_node.cpp)

# Link ROS 2 dependencies
ament_target_dependencies(fleet_manager_node rclcpp std_msgs)

# Install executable to correct location
install(TARGETS
  fleet_manager_node
  DESTINATION lib/${PROJECT_NAME})

# Let ament know we are done
ament_package()
